DELIMITER //


CREATE TRIGGER slug_autor
BEFORE INSERT ON autor
FOR EACH ROW
BEGIN
    DECLARE slug_base VARCHAR(255);
    DECLARE slug_count INT;

    -- Gera o slug básico (nome do autor em formato slug)
    SET slug_base = LOWER(REPLACE(REGEXP_REPLACE(NEW.nome_completo, '[^a-zA-Z0-9 ]', ''), ' ', '-'));

    -- Conta quantos slugs já existem com esse padrão
    SELECT COUNT(*) INTO slug_count FROM autor WHERE slug LIKE CONCAT(slug_base, '%');

    -- Se já existir, adiciona um número incremental ao slug
    IF slug_count > 0 THEN
        SET NEW.slug = CONCAT(slug_base, '-', slug_count + 1);
    ELSE
        SET NEW.slug = slug_base;
    END IF;
END;

//
DELIMITER ;

-- GATILHO PROS SLUGS DE LIVROS
--



DELIMITER //

CREATE TRIGGER slug_livros
BEFORE INSERT ON livros
FOR EACH ROW
BEGIN
    DECLARE slug_base VARCHAR(255);
    DECLARE slug_count INT;
    DECLARE comprimento_maximo INT DEFAULT 100; -- Corresponde ao max_length no modelo Django
    DECLARE primeiras_palavras VARCHAR(100);
    DECLARE slug_autor VARCHAR(100);
    DECLARE nome_autor VARCHAR(255);

    -- Buscar o nome do autor usando o autor_id (chave estrangeira)
    SELECT nome_completo INTO nome_autor FROM autor WHERE id = NEW.autor_id;

    -- Extrair as primeiras 3 palavras do título e substituir espaços por hífens
    SET primeiras_palavras = LOWER(REPLACE(SUBSTRING_INDEX(REGEXP_REPLACE(NEW.titulo, '[^a-zA-Z0-9 ]', ''), ' ', 3), ' ', '-'));

    -- Gerar um slug a partir do nome do autor, substituindo espaços por hífens
    SET slug_autor = LOWER(REPLACE(REGEXP_REPLACE(nome_autor, '[^a-zA-Z0-9 ]', ''), ' ', '-'));

    -- Combinar as primeiras palavras do título e o slug do autor
    SET slug_base = CONCAT(primeiras_palavras, '-', slug_autor);

    -- Garantir que o slug não exceda o comprimento máximo
    SET slug_base = LEFT(slug_base, comprimento_maximo);

    -- Contar quantos slugs já existem com esse padrão
    SELECT COUNT(*) INTO slug_count FROM livros WHERE slug LIKE CONCAT(slug_base, '%');

    -- Se o slug já existir, adicionar um número para torná-lo único
    IF slug_count > 0 THEN
        SET NEW.slug = CONCAT(slug_base, '-', slug_count + 1);
    ELSE
        SET NEW.slug = slug_base;
    END IF;
END;

DELIMITER ;
